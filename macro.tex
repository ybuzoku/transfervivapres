\newdimen\qqArgHgt
\def\Quinequote #1{%
    \setbox\qqBoxA=\hbox{$#1$}%
    \qqArgHgt=\ht\qqBoxA%
    \ifnum     \qqArgHgt<\qqCornerHgt \qqArgHgt=0pt%
    \else \advance \qqArgHgt by -\qqCornerHgt%
    \fi \raise\qqArgHgt\hbox{$\ulcorner$} \box\qqBoxA %
    \raise\qqArgHgt\hbox{$\urcorner$}}
